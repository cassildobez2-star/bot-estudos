async function calcular(){ const modo = document.getElementById("modo").value; const expressao = document.getElementById("entrada").value; const valor = document.getElementById("valorLimite").value; const resposta = await fetch("https://bot-estudos-production.up.railway.app/calcular", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ modo: modo, expressao: expressao, valor: valor }) }); const dados = await resposta.json(); if(dados.resultado){ document.getElementById("resultado").innerText = dados.resultado; } else { document.getElementById("resultado").innerText = dados.erro; } }
